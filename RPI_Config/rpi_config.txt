#RPI Config


#Install Raspian OS on SD card

#Config WiFi:

Save wpa_supplicant.conf file to sd card with

country=US # Your 2-digit country code
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
network={
    ssid="YOUR_NETWORK_NAME"
    psk="YOUR_PASSWORD"
    key_mgmt=WPA-PSK
}


#Allow ssh

Save empty ssh.txt file to sd card

#To ssh

ssh pi@<ip>

default password: raspberry
